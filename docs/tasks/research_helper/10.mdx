<script>
{`
  // Âä®ÊÄÅÂä†ËΩΩÊ®°ÂûãËΩ®ËøπÊï∞ÊçÆ
  async function loadModelTrajectory(modelName) {
    try {
      const response = await fetch(\`/docs/tasks/research_helper/10/\${modelName}.json\`);
      if (!response.ok) {
        console.warn(\`No trajectory data found for \${modelName}\`);
        return null;
      }
      const data = await response.json();
      return data;
    } catch (error) {
      console.warn(\`Error loading trajectory for \${modelName}:\`, error);
      return null;
    }
  }

  // Ëß£ÊûêÂ∑•ÂÖ∑Ë∞ÉÁî®Âπ∂ÁîüÊàêËΩ®ËøπÊ≠•È™§
  function parseTrajectoryData(data) {
    if (!data || !data.messages) return { steps: [], finalOutput: '' };
    
    const steps = [];
    let finalOutput = '';
    
    for (const message of data.messages) {
      if (message.role === 'assistant' && message.tool_calls) {
        for (const toolCall of message.tool_calls) {
          const functionName = toolCall.function.name;
          const args = JSON.parse(toolCall.function.arguments);
          
          // Êü•ÊâæÂØπÂ∫îÁöÑÂ∑•ÂÖ∑ÂìçÂ∫î
          const toolResponse = data.messages.find(msg => 
            msg.role === 'tool' && msg.tool_call_id === toolCall.id
          );
          
          steps.push({
            title: getStepTitle(functionName),
            toolCall: {
              name: functionName,
              args: args
            },
            result: toolResponse ? JSON.parse(toolResponse.content) : null
          });
        }
      } else if (message.role === 'assistant' && message.content && !message.tool_calls) {
        // ÊèêÂèñÊúÄÁªàËæìÂá∫
        const content = message.content;
        if (content.includes('title:') && content.includes('arxiv_abs_url:') && content.includes('code_url:')) {
          finalOutput = content.replace(/\n/g, '<br/>');
        }
      }
    }
    
    return { steps, finalOutput };
  }

  // Ê†πÊçÆÂ∑•ÂÖ∑ÂêçÁß∞ÁîüÊàêÊ≠•È™§Ê†áÈ¢ò
  function getStepTitle(functionName) {
    const titleMap = {
      'arxiv_local-search_papers': 'Search arXiv',
      'arxiv_local-download_paper': 'Download Paper',
      'filesystem-move_file': 'Rename File',
      'filesystem-list_directory': 'Check Files',
      'scholarly-search-arxiv': 'Search Scholarly'
    };
    return titleMap[functionName] || 'Tool Call';
  }

  // Ê∏≤ÊüìËΩ®ËøπÊ≠•È™§
  function renderTrajectorySteps(steps, finalOutput) {
    const container = document.getElementById('trajectory-display');
    if (!container) return;
    
    let html = '';
    
    // Ê∑ªÂä†ÂàùÂßãÊÄùËÄÉÊ≠•È™§
    html += \`
      <div className="trajectory-step">
        <div className="step-header">
          <div className="step-number">1</div>
          <div className="step-title">Initial Analysis</div>
        </div>
        <div className="step-content">
          <div className="thinking">
            <strong>ü§î Thinking:</strong> I'll help you find the paper related to "Alita" and agentic reasoning on arXiv. Let me search for it first.
          </div>
        </div>
      </div>
    \`;
    
    // Ê∏≤ÊüìÂ∑•ÂÖ∑Ë∞ÉÁî®Ê≠•È™§
    steps.forEach((step, index) => {
      html += \`
        <div className="trajectory-step">
          <div className="step-header">
            <div className="step-number">\${index + 2}</div>
            <div className="step-title">\${step.title}</div>
          </div>
          <div className="step-content">
            <div className="tool-call">
              <strong>üîß Tool Call:</strong> \${step.toolCall.name}
              <div className="tool-params">
                <code>\${JSON.stringify(step.toolCall.args, null, 2)}</code>
              </div>
            </div>
            \${step.result ? \`
              <div className="tool-result">
                <strong>üìÑ Result:</strong> \${formatResult(step.result)}
              </div>
            \` : ''}
          </div>
        </div>
      \`;
    });
    
    // Ê∑ªÂä†ÊúÄÁªàËæìÂá∫Ê≠•È™§
    html += \`
      <div className="trajectory-step final">
        <div className="step-header">
          <div className="step-number">‚úì</div>
          <div className="step-title">Task Completed</div>
        </div>
        <div className="step-content">
          <div className="final-output">
            <strong>üéØ Final Output:</strong>
            <div className="output-format">
              <code>\${finalOutput}</code>
            </div>
          </div>
        </div>
      </div>
    \`;
    
    container.innerHTML = html;
  }

  // Ê†ºÂºèÂåñÁªìÊûú
  function formatResult(result) {
    if (typeof result === 'object') {
      if (result.total_results !== undefined) {
        return \`Found \${result.total_results} papers\`;
      }
      if (result.status) {
        return \`Status: \${result.status}\`;
      }
      return JSON.stringify(result).substring(0, 100) + '...';
    }
    return result.toString();
  }

  // Ê®°ÂûãÈÄâÊã©‰∫ã‰ª∂Â§ÑÁêÜ
  document.addEventListener('DOMContentLoaded', function() {
    const modelOptions = document.querySelectorAll('.model-option');
    
    modelOptions.forEach(option => {
      option.addEventListener('click', async function() {
        // ÁßªÈô§ÊâÄÊúâÊ¥ªË∑ÉÁä∂ÊÄÅ
        modelOptions.forEach(opt => opt.classList.remove('active'));
        
        // Ê∑ªÂä†Ê¥ªË∑ÉÁä∂ÊÄÅÂà∞ÂΩìÂâçÈÄâÈ°π
        this.classList.add('active');
        
        // Ëé∑ÂèñÊ®°ÂûãÂêçÁß∞
        const modelName = this.dataset.model;
        
        // Âä†ËΩΩÂπ∂Ê∏≤ÊüìËΩ®ËøπÊï∞ÊçÆ
        const data = await loadModelTrajectory(modelName);
        if (data) {
          const { steps, finalOutput } = parseTrajectoryData(data);
          renderTrajectorySteps(steps, finalOutput);
        }
      });
    });
    
    // ÈªòËÆ§Âä†ËΩΩÁ¨¨‰∏Ä‰∏™Ê®°ÂûãÁöÑÊï∞ÊçÆ
    const firstModel = document.querySelector('.model-option.active');
    if (firstModel) {
      firstModel.click();
    }
  });
`}
</script>
